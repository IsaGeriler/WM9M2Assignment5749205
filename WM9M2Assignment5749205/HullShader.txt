struct HS_INPUT {
	float3 position : POSITION;
};

struct HS_CONTROL_POINT_OUTPUT {
	float3 position : POSITION;
};

struct HS_CONSTANT_DATA_OUTPUT {
	float EdgeTessellation[3] : SV_TessFactor;
	float InsideTessellation : SV_InsideTessFactor;
};

HS_CONSTANT_DATA_OUTPUT HS_PatchConstantFunction(InputPatch<HS_INPUT, 3> patch) {
	HS_CONSTANT_DATA_OUTPUT output;
	output.EdgeTessellation[0] = 5.0f;
	output.EdgeTessellation[1] = 5.0f;
	output.EdgeTessellation[2] = 5.0f;
	output.InsideTessellation = 5.0f;
	return output;
}

[domain("tri")]
[partitioning("fractional_odd")]
[outputtopology("triangle_cw")]
[outputcontrolpoints(3)]
[patchconstantfunc("HS_PatchConstantFunction")]
HS_CONTROL_POINT_OUTPUT 
HS_ControlPointFunction(InputPatch<HS_INPUT, 3> patch, uint
controlPointID : SV_OutputControlPointID) {
	HS_CONTROL_POINT_OUTPUT output;
	output.position = patch[controlPointID].position;
	return output;
}
